Implement Power Saver mode (battery threshold 30%) and enhance the debug overlay with thermal, battery, sample rate, and GPS accuracy rows.

Power Saver reduces IMU capture from 100Hz to 60Hz and GPS update frequency from continuous to duty-cycled (≤1 update per 5s) to extend battery life during a full ski day. The enhanced debug overlay (ClassifierDebugHUD) gives field testers the diagnostics they need to validate system behavior on the mountain.

    ```swift
    private func setupBatteryMonitoring() {
        UIDevice.current.isBatteryMonitoringEnabled = true
        let level = UIDevice.current.batteryLevel
        if level >= 0 { batteryPercent = Int(level * 100) }
        batteryObserver = NotificationCenter.default.addObserver(
            forName: UIDevice.batteryLevelDidChangeNotification,
            object: nil,
            queue: .main
        ) { [weak self] _ in
            Task { @MainActor [weak self] in
                guard let self else { return }
                let level = UIDevice.current.batteryLevel
                guard level >= 0 else { return }
                let pct = Int(level * 100)
                self.batteryPercent = pct
                await self.updatePowerSaverMode(batteryPercent: pct)
            }
        }
    }

    @MainActor
    func updatePowerSaverMode(batteryPercent: Int) async {
        let newMode: PowerSaverMode
        switch powerSaverMode {
        case .normal:
            newMode = batteryPercent <= 30 ? .saving : .normal
        case .saving:
            newMode = batteryPercent >= 35 ? .normal : .saving
        }
        guard newMode != powerSaverMode else { return }
        powerSaverMode = newMode
        await motionManager.setPowerSaverMode(newMode == .saving)
        await gpsManager.setPowerSaverMode(newMode == .saving)
    }
    ```

    ```swift
    func adjustSampleRate(for state: ProcessInfo.ThermalState) {
        let thermalHz: Int
        switch state {
        case .nominal, .fair:   thermalHz = 100
        case .serious:          thermalHz = 50
        case .critical:         thermalHz = 25
        @unknown default:       thermalHz = 50
        }
        // Power saver caps at 60Hz; thermal still wins if it requires lower
        let targetHz = powerSaverEnabled ? min(thermalHz, 60) : thermalHz
        currentSampleRateHz = targetHz
        dataSource.deviceMotionUpdateInterval = 1.0 / Double(targetHz)
    }
    ```

    Update startUpdates to initialize currentSampleRateHz:
    ```swift

        currentRunID = runID

        // ... rest of existing implementation
    ```

    ```swift
    private var lastBroadcastDate: Date = .distantPast
    private let powerSaverMinInterval: TimeInterval = 5.0   // seconds between GPS yields

    func setPowerSaverMode(_ enabled: Bool) {
        powerSaverEnabled = enabled
        if !enabled { lastBroadcastDate = .distantPast }  // reset throttle on disable

    ```

    Update the broadcast method to throttle in power saver mode:
    ```swift
    private func broadcast(_ reading: GPSReading) {
        if powerSaverEnabled {
            let now = reading.timestamp
            guard now.timeIntervalSince(lastBroadcastDate) >= powerSaverMinInterval else { return }
            lastBroadcastDate = now
        }
        for continuation in continuations.values {
            continuation.yield(reading)
        }
    }
    ```
