    ```swift
    // MetricKitSubscriber.swift
    // ArcticEdge
    //
    // Receives MXMetricPayload and MXDiagnosticPayload from MetricKit.
    // Each payload is JSON-serialized and appended to a session log at:
    //   <Documents>/MetricKit/metrics-YYYY-MM-DD.jsonl
    //
    // MetricKit delivers payloads at most once per 24 hours on device.
    // In simulator, payloads never arrive — logging path is never exercised.
    // MXMetricManagerSubscriber must be an NSObject for ObjC protocol conformance.

    import MetricKit
    import Foundation

    final class MetricKitSubscriber: NSObject, MXMetricManagerSubscriber, Sendable {

        override init() {
            super.init()
            MXMetricManager.shared.add(self)
        }

        deinit {
            MXMetricManager.shared.remove(self)
        }

        // MARK: - MXMetricManagerSubscriber

        nonisolated func didReceive(_ payloads: [MXMetricPayload]) {
            for payload in payloads {
                appendPayload(payload.jsonRepresentation(), label: "metric")
            }
        }

        nonisolated func didReceive(_ payloads: [MXDiagnosticPayload]) {
            for payload in payloads {
                appendPayload(payload.jsonRepresentation(), label: "diagnostic")
            }
        }

        // MARK: - Private

        private nonisolated func appendPayload(_ data: Data, label: String) {
            let fm = FileManager.default
            guard let docs = fm.urls(for: .documentDirectory, in: .userDomainMask).first else { return }
            let dir = docs.appendingPathComponent("MetricKit", isDirectory: true)
            try? fm.createDirectory(at: dir, withIntermediateDirectories: true)

            let dateStr = ISO8601DateFormatter().string(from: Date()).prefix(10)
            let file = dir.appendingPathComponent("metrics-\(dateStr).jsonl")

            // Wrap in a simple envelope: {"type":"metric","ts":"...","payload":{...}}
            var envelope: [String: Any] = [
                "type": label,
                "ts": ISO8601DateFormatter().string(from: Date())
            ]
            if let json = try? JSONSerialization.jsonObject(with: data) {
                envelope["payload"] = json
            }
            guard let line = try? JSONSerialization.data(withJSONObject: envelope),
                  var lineStr = String(data: line, encoding: .utf8) else { return }
            lineStr += "\n"

            if fm.fileExists(atPath: file.path) {
                if let handle = try? FileHandle(forWritingTo: file) {
                    handle.seekToEndOfFile()
                    if let append = lineStr.data(using: .utf8) {
                        handle.write(append)
                    }
                    try? handle.close()
                }
            } else {
                try? lineStr.data(using: .utf8)?.write(to: file, options: .atomic)
            }
        }
    }
    ```

    Add to AppModel:
    ```swift
    // MetricKit subscriber — retained for lifetime of app process.
    private let metricKitSubscriber = MetricKitSubscriber()
    ```

    No async setup needed — MetricKitSubscriber registers synchronously in its init.
    Since AppModel.init() is nonisolated, MetricKitSubscriber() can be created there as a
    stored property initializer.

    ```swift
    // CalibrationExporter.swift
    // ArcticEdge
    //
    // Exports one RunRecord and its FrameRecords as a JSON file to:
    //   <Documents>/Calibration/run-<runID>-<date>.json
    //
    // Purpose: Field testers label exported run JSON (skiing vs. chairlift segments)
    // to produce ground-truth data for filter and classifier threshold calibration.
    //
    // Usage:
    //   let exporter = CalibrationExporter(persistence: service)
    //   let url = try await exporter.exportRun(runID: someID)

    import Foundation
    import SwiftData

    actor CalibrationExporter {

        private let persistence: any PersistenceServiceProtocol

        init(persistence: any PersistenceServiceProtocol) {
            self.persistence = persistence
        }

        /// Exports all FrameRecords for `runID` to a JSON file in the Calibration directory.
        /// Returns the file URL on success.
        func exportRun(runID: UUID) async throws -> URL {
            // Fetch frames for this run.
            let frameDescriptor = FetchDescriptor<FrameRecord>(
                predicate: #Predicate { $0.runID == runID },
                sortBy: [SortDescriptor(\.timestamp)]
            )
            let frames = try await persistence.fetchFrameRecords(descriptor: frameDescriptor)

            // Build export payload.
            let payload = CalibrationPayload(
                runID: runID.uuidString,
                exportedAt: ISO8601DateFormatter().string(from: Date()),
                frameCount: frames.count,
                frames: frames.map { CalibrationFrame(record: $0) }
            )

            let data = try JSONEncoder().encode(payload)

            // Write to Documents/Calibration/
            let fm = FileManager.default
            guard let docs = fm.urls(for: .documentDirectory, in: .userDomainMask).first else {
                throw CalibrationExportError.documentsDirectoryUnavailable
            }
            let dir = docs.appendingPathComponent("Calibration", isDirectory: true)
            try fm.createDirectory(at: dir, withIntermediateDirectories: true)

            let datePart = ISO8601DateFormatter().string(from: Date()).prefix(10)
            let fileName = "run-\(runID.uuidString.prefix(8))-\(datePart).json"
            let url = dir.appendingPathComponent(fileName)
            try data.write(to: url, options: .atomic)
            return url
        }
    }

    // MARK: - Export types

    private struct CalibrationPayload: Encodable {
        let runID: String
        let exportedAt: String
        let frameCount: Int
        let frames: [CalibrationFrame]
    }

    private struct CalibrationFrame: Encodable {
        let timestamp: Double
        let pitch: Double
        let roll: Double
        let yaw: Double
        let userAccelX: Double
        let userAccelY: Double
        let userAccelZ: Double
        let filteredAccelZ: Double
        let gpsSpeed: Double?

        init(record: FrameRecord) {
            self.timestamp = record.timestamp
            self.pitch = record.pitch
            self.roll = record.roll
            self.yaw = record.yaw
            self.userAccelX = record.userAccelX
            self.userAccelY = record.userAccelY
            self.userAccelZ = record.userAccelZ
            self.filteredAccelZ = record.filteredAccelZ
            self.gpsSpeed = record.gpsSpeed
        }
    }

    enum CalibrationExportError: Error {
        case documentsDirectoryUnavailable
    }
    ```
