    // LiveViewModel.swift
    // ArcticEdge
    //
    // @Observable @MainActor bridge between StreamBroadcaster and LiveTelemetryView.
    // Maintains a fixed-size waveformSnapshot ([Double] of filteredAccelZ values)
    // for Canvas consumption. Metric values update on every incoming frame.
    // GPS speed is NOT in FilteredFrame; read from appModel.lastGPSSpeed (10Hz HUD).

    import Foundation

    @Observable
    @MainActor
    final class LiveViewModel {

        // MARK: - Public state (read by Canvas and HUD cards)

        private(set) var waveformSnapshot: [Double] = []
        private(set) var pitch: Double = 0
        private(set) var roll: Double = 0
        private(set) var gForce: Double = 0

        // MARK: - Configuration

        let windowSize: Int

        // MARK: - Private

        private var streamTask: Task<Void, Never>?

        // MARK: - Init

        init(windowSize: Int = 1000) {
            self.windowSize = windowSize
        }

        // MARK: - Lifecycle

        // Called from AppModel.startDay() after broadcaster.start().
        // broadcaster is a StreamBroadcaster actor; makeStream() must be awaited.
        func startConsuming(broadcaster: StreamBroadcaster) {
            streamTask?.cancel()
            streamTask = Task { @MainActor [weak self] in
                let stream = await broadcaster.makeStream()
                for await frame in stream {
                    guard let self else { return }
                    // Append filteredAccelZ to ring-buffer snapshot
                    waveformSnapshot.append(frame.filteredAccelZ)
                    if waveformSnapshot.count > windowSize {
                        waveformSnapshot.removeFirst()
                    }
                    // Update HUD metrics
                    pitch = frame.pitch
                    roll = frame.roll
                    gForce = hypot(frame.userAccelX, hypot(frame.userAccelY, frame.userAccelZ))
                }
            }
        }

        func stopConsuming() {
            streamTask?.cancel()
            streamTask = nil
            waveformSnapshot = []
            pitch = 0
            roll = 0
            gForce = 0
        }
    }
    ```

    SWIFT STRICT CONCURRENCY: The Task closure is `@MainActor` — no isolation crossing. `broadcaster.makeStream()` is awaited inside the Task, not in a Canvas closure. waveformSnapshot is read exclusively on @MainActor. No Sendable violations.

    // LiveTelemetryView.swift
    // ArcticEdge
    //
    // Full-screen live telemetry dashboard.
    // Hero: scrolling carve-pressure waveform (~65% screen height) rendered via
    //        TimelineView(.animation) + Canvas at up to 120fps (ProMotion).
    // HUD: four frosted glass metric cards overlaid on the waveform lower quarter.
    // Cursor: fixed right-edge vertical line (oscilloscope "now" marker).
    //
    // Canvas key constraint: draw closure is synchronous. Capture a local immutable
    // copy of waveformSnapshot before the Canvas call — never await or mutate inside.

    import SwiftUI

    struct LiveTelemetryView: View {
        @Environment(AppModel.self) private var appModel
        @State private var liveViewModel: LiveViewModel = LiveViewModel()

        var body: some View {
            ZStack {
                // Same background gradient as ContentView for visual continuity
                backgroundLayer

                VStack(spacing: 0) {
                    // Waveform hero (~65% of available height)
                    waveformSection
                        .frame(maxWidth: .infinity)
                        .frame(maxHeight: .infinity)

                    // Bottom status bar: run elapsed time
                    runStatusBar
                        .frame(height: 64)
                }
            }
            .ignoresSafeArea()
            .onAppear {
                liveViewModel.startConsuming(broadcaster: appModel.broadcaster)
            }
            .onDisappear {
                liveViewModel.stopConsuming()
            }
        }

        // MARK: - Background

        private var backgroundLayer: some View {
            LinearGradient(
                stops: [
                    .init(color: Color(red: 0.051, green: 0.067, blue: 0.090), location: 0),
                    .init(color: Color(red: 0.024, green: 0.039, blue: 0.059), location: 1)
                ],
                startPoint: .top,
                endPoint: .bottom
            )
            .ignoresSafeArea()
        }

        // MARK: - Waveform section

        private var waveformSection: some View {
            ZStack(alignment: .bottomLeading) {
                // Canvas waveform driven by TimelineView at 120fps
                TimelineView(.animation) { _ in
                    // Capture immutable copy before Canvas closure (Sendable safety)
                    let samples = liveViewModel.waveformSnapshot
                    Canvas { context, size in
                        drawWaveform(context: context, size: size, samples: samples)
                        drawCursor(context: context, size: size)
                    }
                }
                .background(Color.black.opacity(0.25))

                // HUD metric cards — float over waveform lower quarter
                hudOverlay
                    .padding(.horizontal, 16)
                    .padding(.bottom, 20)
            }
        }

        // MARK: - Canvas draw functions (called inside Canvas closure — no await allowed)

        private func drawWaveform(context: GraphicsContext, size: CGSize, samples: [Double]) {
            guard samples.count > 1 else { return }
            let xStep = size.width / CGFloat(samples.count - 1)
            let midY = size.height * 0.5
            let scale = size.height * 0.35

            var path = Path()
            path.move(to: CGPoint(x: 0, y: midY - CGFloat(samples[0]) * scale))
            for (i, sample) in samples.dropFirst().enumerated() {
                path.addLine(to: CGPoint(x: CGFloat(i + 1) * xStep,
                                         y: midY - CGFloat(sample) * scale))
            }
            // Arctic blue waveform line
            context.stroke(path,
                           with: .color(Color(red: 0.12, green: 0.56, blue: 1.0).opacity(0.9)),
                           lineWidth: 1.5)
        }

        private func drawCursor(context: GraphicsContext, size: CGSize) {
            // Fixed right-side cursor marks "now"
            let cursorX = size.width - 2
            var cursorPath = Path()
            cursorPath.move(to: CGPoint(x: cursorX, y: 0))
            cursorPath.addLine(to: CGPoint(x: cursorX, y: size.height))
            context.stroke(cursorPath,
                           with: .color(.white.opacity(0.25)),
                           lineWidth: 1)
        }

        // MARK: - HUD overlay

        private var hudOverlay: some View {
            HStack(spacing: 10) {
                LiveMetricCard(
                    label: "PITCH",
                    value: String(format: "%.1f°", liveViewModel.pitch * 180 / .pi)
                )
                LiveMetricCard(
                    label: "ROLL",
                    value: String(format: "%.1f°", liveViewModel.roll * 180 / .pi)
                )
                LiveMetricCard(
                    label: "G-FORCE",
                    value: String(format: "%.2fg", liveViewModel.gForce)
                )
                LiveMetricCard(
                    label: "KM/H",
                    value: appModel.lastGPSSpeed >= 0
                        ? String(format: "%.0f", appModel.lastGPSSpeed * 3.6)
                        : "--"
                )
            }
        }

        // MARK: - Run status bar

        private var runStatusBar: some View {
            HStack {
                // Running indicator dot
                Circle()
                    .fill(Color(red: 0.12, green: 0.56, blue: 1.0))
                    .frame(width: 7, height: 7)
                    .shadow(color: Color(red: 0.12, green: 0.56, blue: 1.0).opacity(0.8), radius: 4)

                Text("LIVE")
                    .font(.system(size: 11, weight: .medium))
                    .tracking(2)
                    .foregroundStyle(Color(red: 0.12, green: 0.56, blue: 1.0))

                Spacer()
            }
            .padding(.horizontal, 24)
            .background(.ultraThinMaterial)
        }
    }

    // MARK: - LiveMetricCard

    private struct LiveMetricCard: View {
        let label: String
        let value: String

        var body: some View {
            VStack(spacing: 6) {
                Text(value)
                    .font(.system(size: 18, weight: .semibold, design: .default))
                    .foregroundStyle(.white)
                    .monospacedDigit()
                    .minimumScaleFactor(0.7)
                Text(label)
                    .font(.system(size: 9, weight: .medium, design: .default))
                    .tracking(2)
                    .foregroundStyle(Color.white.opacity(0.4))
            }
            .frame(maxWidth: .infinity)
            .padding(.vertical, 14)
            .background(.ultraThinMaterial)
            .clipShape(RoundedRectangle(cornerRadius: 12, style: .continuous))
            .overlay(
                RoundedRectangle(cornerRadius: 12, style: .continuous)
                    .strokeBorder(Color.white.opacity(0.07), lineWidth: 0.5)
            )
        }
    }
    ```

    CANVAS ISOLATION RULE: `let samples = liveViewModel.waveformSnapshot` is captured BEFORE the Canvas closure in the TimelineView context (which is @MainActor). The Canvas closure receives a local [Double] — no actor isolation crossing inside the draw closure.

    GPS speed: read from `appModel.lastGPSSpeed` (already on @MainActor, updated at 10Hz by hudPollingTask). Display in km/h (multiply m/s by 3.6). Show "--" when speed == -1 (no GPS fix).

    Do NOT add a preview that requires AppModel() with a real CMMotionManager — use a mock or omit the preview to avoid simulator EXC_BAD_ACCESS (documented Phase 1 decision).
